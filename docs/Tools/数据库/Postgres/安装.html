安装

使用Docker Compose安装​

version: '3.8'

services:
  postgres:
    image: postgres
    container_name: ai_startup_postgres
    environment:
      POSTGRES_USER: postgres
      POSTGRES_PASSWORD: postgres
      POSTGRES_DB: asm
      PGDATA: /data/postgres
    volumes:
      - postgres_data:/data/postgres
    ports:
      - "5432:5432"
    networks:
      - ai_startup_network
    restart: unless-stopped
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U postgres"]
      interval: 10s
      timeout: 5s
      retries: 5

networks:
  ai_startup_network:
    driver: bridge

volumes:
  postgres_data:
