import { interpolate, spring } from 'remotion';

export const fadeIn = (
  frame: number, 
  startFrame: number, 
  duration: number
) => {
  return interpolate(
    frame,
    [startFrame, startFrame + duration],
    [0, 1],
    { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' }
  );
};

export const slideInFromLeft = (
  frame: number,
  startFrame: number,
  duration: number
) => {
  const x = interpolate(
    frame,
    [startFrame, startFrame + duration],
    [-100, 0],
    { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' }
  );
  return `translateX(${x}%)`;
};

export const springScale = (frame: number, fps: number) => {
  return spring({
    frame,
    fps,
    config: { damping: 200 },
  });
};
