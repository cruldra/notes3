зј–иҫ‘еҷЁе®һж—¶е“Қеә”еұһжҖ§еҸҳеҢ–

жҲ‘жңүдёҖдёӘеҲӣе»әжЈӢзӣҳзҡ„и„ҡжң¬пјҢз„¶еҗҺжҲ‘еңЁеңәжҷҜйҮҢйқўдҝ®ж”№иғҢжҷҜиүІпјҢжҲ‘еҫ—жҠҠеңәжҷҜе…іжҺүеҶҚжү“ејҖжүҚеҸҳеҢ–

## жЈӢзӣҳж јеӨ§е°Ҹ (еғҸзҙ )
@export var cell_size: float = 80.0

## жЈӢзӣҳеӨ–иҫ№и·қ (еғҸзҙ )
@export var margin: float = 60.0

## зәҝжқЎйўңиүІ
@export var line_color: Color = Color("#4a3728")

## зәҝжқЎе®ҪеәҰ
@export var line_width: float = 2.0

## иғҢжҷҜйўңиүІ (жЁЎжӢҹжңЁзә№ж·ұеәҰ)
@export var background_color: Color = Color("#dcb35c")

иҝҷжҳҜдёҖдёӘйқһеёёз»Ҹе…ёзҡ„й—®йўҳпјҒиҷҪз„¶дҪ зҡ„д»Јз ҒйҮҢе·Із»ҸжӯЈзЎ®дҪҝз”ЁдәҶ @tool пјҢиҝҷи®©и„ҡжң¬иғҪеңЁзј–иҫ‘еҷЁйҮҢиҝҗиЎҢпјҢдҪҶеҸҳйҮҸдҝ®ж”№еҗҺдёҚдјҡиҮӘеҠЁйҖҡзҹҘ Godot "з”»йқўйңҖиҰҒеҲ·ж–°"гҖӮ

Godot дёәдәҶжҖ§иғҪпјҢдёҚдјҡеӣ дёәеҸҳйҮҸеҖјеҸҳдәҶе°ұиҮӘеҠЁйҮҚз»ҳз•ҢйқўгҖӮдҪ йңҖиҰҒжүӢеҠЁе‘ҠиҜүеј•ж“Һпјҡ"еҳҝпјҢйўңиүІеҸҳдәҶпјҢиҜ·йҮҚж–°жү§иЎҢдёҖж¬Ў _draw() еҮҪж•°гҖӮ"

и§ЈеҶіж–№жі•пјҡдҪҝз”Ё set еҮҪж•° + queue_redraw()вҖӢ

дҪ йңҖиҰҒз»ҷ background_colorпјҲд»ҘеҸҠе…¶д»–еҪұе“ҚеӨ–и§Ӯзҡ„еҸҳйҮҸпјүж·»еҠ дёҖдёӘ setter (и®ҫзҪ®еҷЁ)гҖӮеҪ“дҪ еңЁжЈҖжҹҘеҷЁйҮҢдҝ®ж”№ж•°еҖјж—¶пјҢsetter дјҡиў«и§ҰеҸ‘пјҢжҲ‘д»¬е°ұеңЁйӮЈйҮҢжүӢеҠЁиҜ·жұӮйҮҚз»ҳгҖӮ

иҜ·дҝ®ж”№дҪ зҡ„и„ҡжң¬дёӯеҸҳйҮҸе®ҡд№үзҡ„йғЁеҲҶпјҡ

## иғҢжҷҜйўңиүІ (жЁЎжӢҹжңЁзә№ж·ұеәҰ)
# дҝ®ж”№еүҚпјҡ
# @export var background_color: Color = Color("#dcb35c")

# дҝ®ж”№еҗҺпјҡ
@export var background_color: Color = Color("#dcb35c"):
	set(value):
		background_color = value
		queue_redraw() # <--- е…ій”®пјҒиҝҷеҸҘд»Јз ҒдјҡејәеҲ¶и§ҰеҸ‘ _draw()

дёәд»Җд№Ҳиҝҷж ·еҒҡжңүж•ҲпјҹвҖӢ

  1. set(value): еҪ“дҪ еңЁзј–иҫ‘еҷЁйқўжқҝдҝ®ж”№ background_color ж—¶пјҢGodot дјҡиҮӘеҠЁи°ғз”ЁиҝҷдёӘеҮҪж•°гҖӮ

queue_redraw(): иҝҷдёӘеҮҪж•°дјҡеҗ‘еј•ж“ҺеҸ‘йҖҒдҝЎеҸ·пјҢж Үи®°еҪ“еүҚиҠӮзӮ№дёә "и„ҸпјҲdirtyпјү" зҠ¶жҖҒгҖӮGodot дјҡеңЁдёӢдёҖеё§иҮӘеҠЁйҮҚж–°и°ғз”Ё _draw() пјҢд»ҺиҖҢз”Ёж–°зҡ„йўңиүІз»ҳеҲ¶зҹ©еҪўгҖӮ

е»әи®®пјҡжҠҠжүҖжңүи§Ҷи§үеҸҳйҮҸйғҪеҠ дёҠвҖӢ

дёәдәҶи®©дҪ и°ғж•ҙ cell_size гҖҒmargin жҲ– line_width ж—¶д№ҹиғҪе®һж—¶зңӢеҲ°еҸҳеҢ–пјҢе»әи®®жҠҠе®ғд»¬д№ҹйғҪж”№жҲҗиҝҷз§ҚеҶҷжі•пјҡ

@export var cell_size: float = 80.0:
	set(v):
		cell_size = v
		queue_redraw()

@export var margin: float = 60.0:
	set(v):
		margin = v
		queue_redraw()

# ... е…¶д»–еҸҳйҮҸеҗҢзҗҶ

иҝҷж ·дҝ®ж”№еҗҺпјҢдҪ еңЁзј–иҫ‘еҷЁйҮҢжӢ–еҠЁйўңиүІжҲ–ж»‘еқ—пјҢжЈӢзӣҳе°ұдјҡе®һж—¶еҲ·ж–°пјҢе®Ңе…ЁдёҚйңҖиҰҒйҮҚеҗҜеңәжҷҜгҖӮ
